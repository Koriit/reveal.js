        public fun <E> Channel(capacity: Int = RENDEZVOUS): Channel<E> =
            when (capacity) {
     /*  0 */   RENDEZVOUS -> RendezvousChannel()
     /* -1 */   UNLIMITED -> LinkedListChannel()
     /* -2 */   CONFLATED -> ConflatedChannel()
     /* -3 */   BUFFERED -> ArrayChannel(CHANNEL_DEFAULT_CAPACITY)
                else -> ArrayChannel(capacity)
            }
