PATCH Example - Patches

    class UserPatch : PatchOf<User>() {
        var name by patchOf(User::name)
        var age by patchOf(User::age)
        var address by patchOf(User::address, by = AddressPatch::class)
    }
    

    class AddressPatch : PatchOf<Address>() {
        var line1 by patchOf(Address::line1)
        var line2 by patchOf(Address::line2)
        var line3 by patchOf(Address::line3)
    }
