Configuration

    bind<HttpClient>() with provider {
        HttpClient {
            install(DefaultRequest) {
                header(HttpHeaders.XRequestId, correlationId)
            }
            install(ClientLogging) {
                logFullUrl = true
                logHeaders = true
                logBody = true
            }
            install(JsonFeature) {
                serializer = JacksonSerializer(instance())
            }
        }
    }
