        // (no Correlation)

        // (Correlation #1)

            correlateThread(correlationId)
            // (Correlation #1)
        }.run()

        // (Correlation #1)

        Thread {
            // (no Correlation)
            subCorrelateThread(correlationId)
            // (Correlation #1, SubCorrelation #1)
        }.run()

        // (Correlation #1)

        Thread {
            // (no Correlation)
            correlateThread()
            // (Correlation #2)
        }.run()

        // (Correlation #1)

    }
