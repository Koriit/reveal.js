Jobs
  • Jobs provide coroutine status
  • Allow cancelling(coroutine must be cancellable)
  • Allow building parent-child hierarchy of coroutines(structured concurrency)
  • Exceptions from child coroutines propagate to parent(cancellation too)

        suspend fun concurrentSum(): Int = coroutineScope {
            val one = async { doSomethingUsefulOne() }
            val two = async { doSomethingUsefulTwo() }
            one.await() + two.await()
        }
